155 __ix32 FreqCharacteristics( __type F )
156 {
157 __type Re_nom = 0 , Im_nom = 0 , Re_den = 0 , Im_den = 0 , Re = 0 , Im = 0;
158 Re_nom = 1 - cos(-PI2 * F * m_Ns * m_Ts);
159 Im_nom = 0 - sin(-PI2 * F * m_Ns * m_Ts);
160 Re_den = 1 - cos(-PI2 * F * m_Ts);
161 Im_den = 0 - sin(-PI2 * F * m_Ts);
162 Re = (Re_nom * Re_den + Im_nom * Im_den) / (Re_den * Re_den + Im_den * Im_den);
163 Im = (Re_den * Im_nom - Re_nom * Im_den) / (Re_den * Re_den + Im_den * Im_den);
164 m_Km = m_Gain * sqrt(Re * Re + Im * Im);
165 m_pH = atan2(Im , Re);
166 return 0;
167 }
