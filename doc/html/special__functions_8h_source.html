212extern __fx64 __am__( __fx64 u , __fx64 k )
213{
214 __fxx64 a[__ITERATIONS_NUMBER__ + 1];
215 __fxx64 g[__ITERATIONS_NUMBER__ + 1];
216 __fxx64 c[__ITERATIONS_NUMBER__ + 1];
217 __fxx64 two_n;
218 __fxx64 phi;
219 __ix32 n;
220
221 k = fabsl((__fxx64)k);
222
223 // if k is out of range [-1 ; +1] we should set an edge value...
224 if ( k < -1 ) k = -1;
225 else if( k > +1 ) k = +1;
226
227 // special cases check: k = 1 and x = 0
228 if (k == 0.0) return u;
229 if (k == 1.0) return 2.0 * atan(exp(u)) - PI2;
230
231 // arithmetico-geometric mean method variables initialization:
232 a[0] = 1.0L;
233 g[0] = sqrtl(1.0L - k * k);
234 c[0] = k;
235
236 /*
237 Gauss arithmetic operations are continued untill the sequence converges to
238 the answer that satisfies machinery epsilon accuracy
239 */
240
241 two_n = 1.0L;
242
243 for (n = 0; n < __ITERATIONS_NUMBER__; n++)
244 {
245 if (fabsl(a[n] - g[n]) < (a[n] * __EPSILON_xx64__)) break;
246 two_n += two_n;
247 a[n + 1] = 0.5L * (a[n] + g[n]);
248 g[n + 1] = sqrtl(a[n] * g[n]);
249 c[n + 1] = 0.5L * (a[n] - g[n]);
250 }
251
252 // backward substitution implementation:
253 phi = two_n * a[n] * u;
254 for (; n > 0; n--) phi = 0.5L * (phi + asinl(c[n] * sinl(phi) / a[n]));
255 return ( __fx64 )phi;
256}

367 // initialization:
368 k = fabs(x);
369 a[0] = 1.0L;

371
372 // forward substitution:
373 two_n = 1.0L;
374 for (n = 0; n < __ITERATIONS_NUMBER__ - 1; n++)
375 {

378 a[n + 1] = 0.5 * (a[n] + g[n]);

384 for (n = 1 ; n < __ITERATIONS_NUMBER__ ; n++)
