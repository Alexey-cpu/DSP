207extern __fx64 __am__( __fx64 u , __fx64 k )
208{
209 __fxx64 a[__ITERATIONS_NUMBER__ + 1];
210 __fxx64 g[__ITERATIONS_NUMBER__ + 1];
211 __fxx64 c[__ITERATIONS_NUMBER__ + 1];
212 __fxx64 two_n;
213 __fxx64 phi;
214 __ix32 n;
215
216 k = fabsl((__fxx64)k);
217
218 // if k is out of range [-1 ; +1] we should set an edge value...
219 if ( k < -1 ) k = -1;
220 else if( k > +1 ) k = +1;
221
222 // special cases check: k = 1 and x = 0
223 if (k == 0.0) return u;
224 if (k == 1.0) return 2.0 * atan(exp(u)) - PI2;
225
226 // arithmetico-geometric mean method variables initialization:
227 a[0] = 1.0L;
228 g[0] = sqrtl(1.0L - k * k);
229 c[0] = k;
230
231 /*
232 Gauss arithmetic operations are continued untill the sequence converges to
233 the answer that satisfies machinery epsilon accuracy
234 */
235
236 two_n = 1.0L;
237
238 for (n = 0; n < __ITERATIONS_NUMBER__; n++)
239 {
240 if (fabsl(a[n] - g[n]) < (a[n] * __EPSILON_xx64__)) break;
241 two_n += two_n;
242 a[n + 1] = 0.5L * (a[n] + g[n]);
243 g[n + 1] = sqrtl(a[n] * g[n]);
244 c[n + 1] = 0.5L * (a[n] - g[n]);
245 }
246
247 // backward substitution implementation:
248 phi = two_n * a[n] * u;
249 for (; n > 0; n--) phi = 0.5L * (phi + asinl(c[n] * sinl(phi) / a[n]));
250 return ( __fx64 )phi;
251}

356{

359 __fx64 two_n , phi , phi_old , k;
360 __ix32 n = 0;
361
362 // initialization:
363 k = fabs(x);
364 a[0] = 1.0L;
365 g[0] = sqrt(1.0 - k * k);
366
367 // forward substitution:
368 two_n = 1.0L;
369 for (n = 0; n < __ITERATIONS_NUMBER__ - 1; n++)
370 {
371 s[n] = 0;
372 two_n += two_n;
373 a[n + 1] = 0.5 * (a[n] + g[n]);
374 g[n + 1] = sqrt(a[n] * g[n]);
375 }
376
377 // backward substitution:
378 phi = asin(sn);
379 for (n = 1 ; n < __ITERATIONS_NUMBER__ ; n++)
380 {
381

390 {
