221 // complex frequency coeffs:
222 __type Re1 = 0 , Im1 = 0 , Re2 = 0 , Im2 = 0 , K = 1 / m_Ns;
223
224 // transfer function:
225 Re1 = 1 - cos( -PI2 * F * m_Ns * m_Ts );
226 Im1 = 0 - sin( -PI2 * F * m_Ns * m_Ts );
227 Re2 = 1 - ( cos(-PI2 * F * m_Ts) * m_Kc - sin(-PI2 * F * m_Ts) * m_Ks );
228 Im2 = 0 - ( cos(-PI2 * F * m_Ts) * m_Ks + sin(-PI2 * F * m_Ts) * m_Kc );

232 {
233 m_Km = 1;
234 m_pH = 0;
235 }
236 else
