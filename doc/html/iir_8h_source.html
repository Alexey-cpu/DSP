358

360 __fx64 epsilon_stop = sqrt(pow(10, g_stop / 10) - 1);
361 __fx64 epsilon_pass = sqrt(pow(10, g_pass / 10) - 1);

363 // identify the number of zeros and poles:
364 __ix32 L = trunc( order / 2 ) , R = order - 2 * L , N = L + R;

374

384 KE = __ellip_k__( m );
385 SN = __sn__( alpha * KE , m );
386 Kp = Kp * SN*SN*SN*SN;
387 }

391
392 // ZEROS COMPUTATION:
393
394 KE = __ellip_k__( Kw );
395 if( order % 2 == 0 ) // even order filter
396 {
397 for ( __ix32 i = 0; i < L; i++)
398 {
399 alpha = (2 * i + 1) / ( ( T )order) * KE;
400 SN = __sn__( alpha , Kw );
401 zlp[i] = complex< T >( 0 , 1 /( Kw * SN ) );
402 }
403 }
404 else if( order % 2 != 0 ) // odd order filter
405 {
406 for ( __ix32 i = 0; i < L; i++)
407 {
408 alpha = (2 * i + 2) / ( ( T )order) * KE;
409 SN = __sn__( alpha , Kw );
410 zlp[i] = complex< T >( 0 , 1 /( Kw * SN ) );
411 }
412 }
413
414 // POLES COMPUTATION:
415 T V0 = 0 , A = 0 , B = 0 , C = 0 , D = 0 , E = 0 , F = 0;
416 V0 = -__ellip_k__( Kw ) * __isc__(1 / epsilon_pass, sqrt(1 - Ke * Ke)) / ( __ellip_k__( Ke ) * ( ( T ) order ) );
417 KE = __ellip_k__( Kw );
418
419 // complex conjugate pairs:
420 for ( __ix32 i = 0; i < L; i++)
421 {
422 if( order % 2 == 0 ) // even order filter
423 {
424 alpha = (2 * i + 1) / ( ( T ) order ) * KE;
425 }
426 else // odd order filter
427 {
428 alpha = (2 * i + 2) / ( ( T ) order ) * KE;
429 }
430

432 A = __cn__( alpha , Kw );
433 B = __dn__( alpha , Kw );
434 C = __sn__( V0, sqrt( 1 - Kw * Kw));
435 D = sqrt(1 - C * C);
436 E = B * B;
437 F = C * C;
438
439 // pole real part:
440 plp[i].m_re = A * B * C * D / (1 - E * F);
441
442 // compute auxiliary variables:
443 A = __sn__( alpha, Kw );
444 B = __dn__( V0 , sqrt(1 - Kw * Kw ) );
445 C = __dn__( alpha, Kw );
446 D = __sn__( V0, sqrt( 1 - Kw * Kw ) );
447 E = C * C;
448 F = D * D;
449
450 // pole imaginary part:
451 plp[i].m_im = A * B / (1 - E * F);
452 }
453
454 // real odd pole:
455 if( R >= 1 )
456 {
457 // compute auxiliary variables:
458 A = __sn__( V0 , sqrt( 1 - Kw * Kw ) );
459 B = __cn__( V0 , sqrt( 1 - Kw * Kw ) );
460 C = A * A;
461
462 // compute the real odd pole:
463 plp[ N - 1 ] = complex< T >( A * B / ( 1 - C ) , 0 );
464 }
465
466 // GAINS COMPUTATION:
467
468 // complex conjugate pairs:
469 if( R >= 1 )
470 {
471 // complex conjugate pairs:
472 for ( __ix32 i = 0; i < L; i++)
473 {
474 glp[i] = ( zlp[i] * __conjf__( zlp[i] ) ) / ( plp[i] * __conjf__( plp[i] ) );
475 glp[i] = complex< T >( 1 , 0 ) / glp[i];
476 }
477
478 // real odd pole:
479 glp[ N - 1 ] = -plp[ N - 1 ].m_re;
480
481 // output gain:
482 glp[N] = 1;
483 }
484 else
485 {
486 T a = sqrt( 1 / (1 + epsilon_stop * epsilon_stop) );
487 glp[N] = pow ( a , 1 / ( ( T ) L ) );
488 for ( __ix32 i = 0; i < L; i++) glp[i] = glp[N];
489 }
490
491 return iir_zp< T >{ plp , zlp , glp , L , R , N };
492}
