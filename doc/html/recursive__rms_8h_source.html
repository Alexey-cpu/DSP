156 __ix32 FreqCharacteristics( __type F )
157 {
158 __type Re_nom = 0 , Im_nom = 0 , Re_den = 0 , Im_den = 0 , Re = 0 , Im = 0;
159 Re_nom = 1 - cos(-PI2 * F * m_Ns * m_Ts);
160 Im_nom = 0 - sin(-PI2 * F * m_Ns * m_Ts);
161 Re_den = 1 - cos(-PI2 * F * m_Ts);
162 Im_den = 0 - sin(-PI2 * F * m_Ts);
163 Re = (Re_nom * Re_den + Im_nom * Im_den) / (Re_den * Re_den + Im_den * Im_den);
164 Im = (Re_den * Im_nom - Re_nom * Im_den) / (Re_den * Re_den + Im_den * Im_den);
165 m_Km = m_Gain * sqrt(Re * Re + Im * Im);
166 m_pH = atan2(Im , Re);
167 return 0;
168 }
