124template< typename OT , typename CT > fir_fr<OT> __fir_freq_resp__( __fx64 Fs , __fx64 F , __ix32 order , CT *cfbuff )
125{
126 OT Re = 0 , Im = 0 , Ts = 1 / Fs;
127 for ( __ix32 i = 0; i <= order; i++)
128 {
129 Re = Re + cos(-PI2 * i * F * Ts) * cfbuff[ i ];
130 Im = Im + sin(-PI2 * i * F * Ts) * cfbuff[ i ];
131 }
132 OT Km = sqrt (Re * Re + Im * Im) , pH = atan2( Im , Re );
133 return { Km , pH };
134}
